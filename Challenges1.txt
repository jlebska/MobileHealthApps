fun main() {
    val number: Int = 12
    var counter:Double=0.0
    val odds:Double

    for(dice1 in 1 until 7){
        for(dice2 in 1 until 7){
            if(dice1+dice2==number){
                counter+=1.0
            }
        }
    }
    odds=counter/36
    println("The odds of rolling the number $number are $odds")
}
